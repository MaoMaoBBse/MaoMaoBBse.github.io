装饰模式（Decorator）

作用

动态的为已有对象添加新功能，相对于生成子类来说更灵活。通过新增功能对原有对象进行装饰，从而名为装饰模式。

结构图

具体实现

基类

// 基类
class Base {
public:
  using ptr = std::shared_ptr<Base>;	// 基类指针
  virtual void operation() = 0;
};

派生类

// 派生类
class Derived : public Base {
public:
  using ptr = std::shared_ptr<Derived>; // 派生类指针
  virtual void operation() override { std::cout << "具体对象的操作\n"; }
};

新增功能类

// 新增功能类/装饰类
class Decorator : public Base {
public:
  using ptr = std::shared_ptr<Decorator>; // 装饰类指针

	// 设置需要装饰的对象
	void setBase(Base::ptr base_ptr) {
    m_base_ptr = base_ptr;
  }

	// 重写操作函数，当设置了装饰的对象时调用该对象的操作
  void operation() override {
    if (m_base_ptr) 
      m_base_ptr->operation();
  }

private:
  Base::ptr m_base_ptr{};
};

功能1类

// 具体功能1
class ConcreteDecrator1 : public Decorator {
public:
  using ptr = std::shared_ptr<ConcreteDecrator1>; // 功能1指针
  void operation() override {
    Decorator::operation();
    m_added_data = "New data";
    std::cout << "具体装饰对象1的操作\n";
  }

private:
  std::string m_added_data{};  // 新增内容由于区分功能2
};

功能2类

// 具体功能2
class ConcreteDecrator2 : public Decorator {
public:
  using ptr = std::shared_ptr<ConcreteDecrator2>;	// 功能2指针
  void operation() override {
    Decorator::operation();
    addedBehavior();
    std::cout << "具体装饰对象2的操作\n";
  }

private:
  void addedBehavior() {}
};

测试代码

auto derived_ptr{std::make_shared<Derived>()};
 auto decrator1_ptr{std::make_shared<ConcreteDecrator1>()};
 auto decrator2_ptr{std::make_shared<ConcreteDecrator2>()};

 // 为派生类装饰新功能1
 std::cout << "为派生类装饰新功能1:\n";
 decrator1_ptr->setBase(derived_ptr);
 decrator1_ptr->operation();
 std::cout << "-------------------------\n";

 // 为派生类装饰新功能2
 std::cout << "为派生类装饰新功能2:\n";
 decrator2_ptr->setBase(derived_ptr);
 decrator2_ptr->operation();
 std::cout << "-------------------------\n";

 // 在为已经装饰好功能1的基础上，再装饰个功能2
 std::cout << "在为已经装饰好功能1的基础上，再装饰个功能2\n";
 decrator2_ptr->setBase(decrator1_ptr);
 decrator2_ptr->operation();
 std::cout << "-------------------------\n";

总结

装饰模式是能在已有功能的基础上动态的添加新功能的一种方式。当新加入的功能只是在特定情况下时才会使用，此时装饰模式就是不错的选择。
